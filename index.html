📶 Low-Bandwidth Architecture
• Per-record PATCH — each scan sends only ~200 bytes to scanned/{id}, never a full dump
• Write queue — rapid scans are batched locally and flushed in a single burst (600ms debounce)
• SSE stream — admin receives push notifications; no polling overhead on table devices
• Offline first — all writes saved to localStorage instantly; synced automatically on reconnect
• Retry logic — failed writes retry with exponential back-off (400ms → 800ms → 1.6s)

function doGet(e){
  var ss=SpreadsheetApp.getActiveSpreadsheet();
  var sheetParam=e&&e.parameter&&e.parameter.sheet?e.parameter.sheet:'';
  var action=e&&e.parameter&&e.parameter.action?e.parameter.action:'data';

  var sheetNames=sheetParam?sheetParam.split(',').map(function(s){return s.trim();}).filter(Boolean):[];
  var sheets=sheetNames.length
    ? sheetNames.map(function(n){return ss.getSheetByName(n);}).filter(Boolean)
    : [ss.getActiveSheet()];

  if(!sheets.length){
    var avail=ss.getSheets().map(function(s){return s.getName();}).join(', ');
    var out=ContentService.createTextOutput(JSON.stringify({error:'None of the requested sheets found. Available: '+avail}));
    out.setMimeType(ContentService.MimeType.JSON);return out;
  }

  if(action==='hash'){
    var hashParts=[];
    sheets.forEach(function(sh){
      var vals=sh.getDataRange().getValues();
      var rowCount=vals.length-1;
      var tail='';var start=Math.max(1,vals.length-3);
      for(var i=start;i<vals.length;i++)tail+=vals[i].join('|');
      hashParts.push(sh.getName()+':'+rowCount+':'+tail.length+':'+tail.slice(0,80));
    });
    var out=ContentService.createTextOutput(JSON.stringify({hash:hashParts.join('||'),rows:hashParts.length,ts:Date.now()}));
    out.setMimeType(ContentService.MimeType.JSON);return out;
  }

  var allRows=[];
  sheets.forEach(function(sh){
    var range=sh.getDataRange();
    var data=range.getValues();
    var disp=range.getDisplayValues();
    var headers=data[0];
    var poIdx=-1,invIdx=-1;
    for(var j=0;j<headers.length;j++){
      var h=String(headers[j]).toLowerCase().replace(/[\s._-]+/g,'');
      if(h==='pono'||h==='ponumber'||h==='purchaseorderno'||h==='po')poIdx=j;
      if(h==='invoiceno'||h==='invno'||h==='invoicenumber'||h==='invoice#'||h==='inv#')invIdx=j;
    }
    for(var i=1;i<data.length;i++){
      if(!data[i][0]&&!data[i][1])continue;
      var row={_sourceSheet:sh.getName()};
      for(var j=0;j<headers.length;j++){
        row[headers[j]]=(j===poIdx||j===invIdx)?disp[i][j]:data[i][j];
      }
      allRows.push(row);
    }
  });
  var out=ContentService.createTextOutput(JSON.stringify({sheets:sheets.map(function(s){return s.getName();}),data:allRows}));
  out.setMimeType(ContentService.MimeType.JSON);
  return out;
}
